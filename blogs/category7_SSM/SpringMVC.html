    # 区别

    • @RequestParam用于接收url地址传参或表单传参
    • @RequestBody用于接收json数据
    • @PathVariable用户接收路径参数，使用{参数名称}描述路径参数

    # 应用

    • 后期开发中，发送请求参数超过1小时，以json格式为主，@RequestBody应用较广
    • 非json格式数据，选用@RequestParam接收请求参数
    • 参数个数较少，使用@PathVariable，通常用语传递id值

    @RestController
@RequestMapping("/books")
public class BookController {
    @Autowired
    private BookService bookService;

    @PostMapping
    public JsonData save(@RequestBody Book book) {
        boolean flag =  bookService.save(book);
        return new JsonData(flag ? Code.SAVE_OK : Code.SAVE_ERR,flag);
    }

    @PutMapping
    public JsonData update(@RequestBody Book book) {
        boolean flag = bookService.update(book);
        return new JsonData(flag ? Code.UPDATE_OK : Code.UPDATE_ERR,flag);
    }

    @DeleteMapping("/{id}")
    public JsonData delete(@PathVariable Integer id) {
        boolean flag = bookService.delete(id);
        return new JsonData(flag ? Code.DELETE_OK : Code.DELETE_ERR,flag);
    }

    @GetMapping("/{id}")
    public JsonData getByid(@PathVariable Integer id) {
        Book book = bookService.getByid(id);
        Integer code = book != null ? Code.GET_OK : Code.GET_ERR;
        String msg = book != null? "":"数据查询失败，请重试";
        return new JsonData(code,book,msg);
    }

    @GetMapping
    public JsonData getAll() {
        List<Book> list =  bookService.getAll();
        Integer code = list != null ? Code.GET_OK : Code.GET_ERR;
        String msg = list != null? "":"数据查询失败，请重试";
        return new JsonData(code,list,msg);
    }
}

    # 异常处理器

    框架内部抛出的异常:因使用不合规导致

    数据层抛出的异常:因外部服务器故障导致（例如:服务器访问超时)

    业务层抛出的异常:因业务逻辑书写错误导致（例如:遍历业务书写操作，导致索引异常等)

    表现层抛出的异常:因数据收集、校验等规则导致（例如:不匹配的数据类型间导致异常)

    工具类抛出的异常:因工具类书写不严谨不够健壮导致（例如:必要释放的连接长期未释放等)

    将所有异常抛出到表现层进行处理

    # 项目异常分类

    • 业务异常businessException
      • 规范的用户行为产生的异常
      • 不规范的用户行为操作产生的异常
    • 系统异常systemException
      • 项目运行过程中可预计且无法避免的异常
    • 其他异常exception
      • 编程人员未预期到的异常

    # 处理方案

    • 业务异常
      • 发送对应消息传递给用户，提醒操作规范
    • 系统异常
      • 发送特定消息传递给用户
      • 发送特定消息传递给相关人员
      • 记录日志
    • 其他异常
      • 发送特定消息传递给用户
      • 发送特定消息传递给编程人员，维护
      • 记录日志

    public class SystemException extends RuntimeException {
    private Integer Code;//加入异常编号，便于识别

    public SystemException(Integer code) {
        Code = code;
    }
    public SystemException(Integer code, String message) {
        super(message);
        Code = code;
    }
    public SystemException(Integer code, String message, Throwable cause) {
        super(message, cause);
        Code = code;
    }
    public Integer getCode() {
        return Code;
    }
    public void setCode(Integer code) {
        Code = code;
    }
}
