    public static void Bubble02(int[] arr, Comparator c) {
        int temp1 = 0;
        for (int i = 0; i < arr.length - 1; i++) {
            for (int j = 0; j < arr.length - i - 1; j++) {
//数组排序由c.compare(arr[j],arr[j+1])返回的值决定
                if (c.compare(arr[j], arr[j + 1]) > 0) {
                    temp1 = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp1;
                }
            }
        }
    }
}

    1. binarySearch通过二分搜索法进行查找，要求必须排好序
      int index二Arrays.binarySearch(arr, 3);
    public class ArraysMethod02 {
    public static void main(String[] args) {
        Integer[] arr = {1,2,3,5,7,8};
        //binarySearch通过二分搜索法进行查找，要求必须排好序
        //1.使用binarySearch 二叉查找
        //2.要求数组有序，如果数组无序，不能使用binarySearch
        //3.如果数组不存在该元素，就返回 -(low+1)  //key not found
        int index = Arrays.binarySearch(arr,4);
        System.out.println("index="+index);
    }
}
