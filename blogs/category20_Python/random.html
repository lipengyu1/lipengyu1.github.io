    # random.choice() 和 random.sample()

    random.choice和random.sample都是Python中random模块提供的函数，用于从给定的序列中随机选择元素。

    random.choice(sequence)函数从一个非空的序列中随机选择一个元素，并返回该元素。这个序列可以是一个列表、元组、字符串或其他可迭代对象。

    random,sample(oooulaton,k)函数从指定的序列oopulation中随机选择K个不重复的元素，并以列表的形式返回，如果指定的k大于席列的长度，则会引发ValueError异常。

    import random
from tabulate import tabulate

random.seed(10)

table_header = ['id','title','results','answer']
table_data = []
for i in range(1,11):
    a = random.randint(1,101)
    b = random.randint(1,101)
    c = random.choice(["+","-","*","/"])
    ans = round(eval(f"{a}{c}{b}"),2)
    table_data.append([i,f"{a} {c} {b} = ","",ans])

print(tabulate(table_data, headers=table_header, tablefmt='grid'))

    # 随机密码

    import random
import string

random.seed(10)


def random_password():
    resource = string.ascii_letters+string.digits+string.punctuation+string.ascii_uppercase
    password = random.sample(resource,6)
    password += random.sample(string.ascii_uppercase,2)
    password += random.choice(string.digits)
    password += random.choice(string.punctuation)
    # password_list = list(password)
    # password = ''.join(password_list)
    password = ''.join(password)
    return password


if __name__ == '__main__':
    print(random_password())

    # random.shuffle

    用于将一个列表中的元素打乱，随机排序

    import random
p=['hehe','xixi','heihei','haha','zhizhi','lala','momo..da']
random.shuffle(p)
print(p)
# ['hehe', 'xixi', 'zhizhi', 'lala', 'heihei', 'momo..da', 'haha']
x = [1, 2, 3, 4, 5]
random.shuffle(x)
print(x)
# [2, 5, 3, 1, 4]
