class Queue():
    def __init__(self):
        self.queue = []

    def enqueue(self, item):
        self.queue.append(item)  # 添加

    def dequeue(self):
        if not self.is_empty():
            return print("T")
        else:
            return print("F")

    def peek(self):
        return self.queue[0]

    def is_empty(self):
        return len(self.queue) == 0

    def size(self):
        return len(self.queue)

    # 读取文件数据，封装到list

class TextFileReader(FileReader):
    def __init__(self, path):
        self.path = path  # 定义成员变量，记录文件路径

    # 实现抽象方法
    def read_data(self) -> list[Record]:
        file = open(self.path, 'r', encoding='utf-8')
        recode_list = []
        try:
            for line in file.readlines():
                line = line.strip()  # 删除换行符\n、空格  2011-01-31,7f3e44b1-ac69-4540-9dfe-85fa83bd3841,207,贵州省
                data_list = line.split(',')  # 分割  ['2011-01-31', '7f3e44b1-ac69-4540-9dfe-85fa83bd3841', '207', '贵州省']
                recode = Record(data_list[0], data_list[1], int(data_list[2]), data_list[3])
                recode_list.append(recode)
        except FileNotFoundError:
            print("FileNotFoundE")
        finally:
            file.close()
        return recode_list


class JsonFIleReader(FileReader):

    def __init__(self, path):
        self.path = path

    def read_data(self) -> list[Record]:
        file = open(self.path, 'r', encoding='utf-8')
        recode_list = []
        try:
            for line in file.readlines():
                data_dict = json.loads(line)
                record = Record(data_dict["date"], data_dict["order_id"], int(data_dict["money"]),
                                data_dict["province"])
                recode_list.append(record)
        except FileNotFoundError:
            print("FileNotFoundE")
        finally:
            file.close()
        return recode_list
