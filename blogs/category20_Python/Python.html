    • 数字精度控制 m.n

    m控制宽度，要求是数字，设置的宽度小于数字自身，不生效

    .n控制小数点精度，进行小数四舍五入

    >>> num1 = 11
>>> num2 = 11.345
>>> print("数字11宽度限制5，结果是：%5d"% num1)
# 数字11宽度限制5，结果是：   11
>>> print("数字11宽度限制5，结果是：%1d"% num1)
# 数字11宽度限制5，结果是：11
>>> print("数字11.345宽度限制7,小数精度2，结果是：%7.2f"% num2)
# 数字11.345宽度限制7,小数精度2，结果是：  11.35
>>> print("数字11.345,小数精度2，结果是：%.2f"% num2)
# 数字11.345,小数精度2，结果是：11.35

    # 练习

    money = 5000
def menu():
    print("________操作________")
    print("1查询")
    print("2存")
    print("3取")
    print("4退出")
    num = input("请输入操作")
    return num
def sel(show):
    if show:
        print("_______余额查询_______")
    print(f"剩余{money}")
def into(mo):
    global money
    money += mo
    print("_______存钱_______")
    print(f"存{mo}")
    sel(False)
def getout(mo):
    print("_______取钱_______")
    global money
    money -= mo
    if money>=0:
        print(f"取{mo}")
    else:
        money+=mo
        print("没钱")
    sel(False)
while True:
    num = menu()
    if num == "1":
        sel(True)
        continue
    elif num == "2":
        x = int(input("存入"))
        into(x)
        continue
    elif num == "3":
        x = int(input("取出"))
        getout(x)
        continue
    elif num == "4":
        break
    else:
        print("重新输入")
