    # 线性回归

    '''
线性回归
'''
import numpy as np
import matplotlib.pyplot as plt

# 读取文件
train = np.loadtxt('click.csv', delimiter=',', skiprows=1)
train_x = train[:, 0]  # 取出第一列
train_y = train[:, 1]  # 取出第二列
# plt.plot(train_x, train_y, "o")
# plt.show()

# 标准化(z-score规范化)
# 把训练数据变成平均值为0、方差为 1 的数据,加快参数收敛
mu = train_x.mean()  # mean() 给定数组的平均值
sigma = train_x.std()  # std() 标准差


# 标准化
def standardize(x):
    return (x - mu) / sigma


train_z = standardize(train_x)
# plt.plot(train_z, train_y, "o")
# plt.show()

# 作为一次函数实现
# 参数初始化
theta0 = np.random.rand()
theta1 = np.random.rand()


# 预测函数
def f(x):
    return theta0 + theta1 * x


# 目标函数
def E(x, y):
    return 0.5 * np.sum(y - f(x)) ** 2


# 学习率10^-3
ETA = 1e-3

# 误差的差值
diff = 1

# 更新次数
count = 0

# 重复学习
error = E(train_z, train_y)  # 误差
while diff > 1e-2:
    # 更新结果保存到临时变量
    tmp0 = theta0 - ETA * np.sum(f(train_z) - train_y)
    tmp1 = theta1 - ETA * np.sum((f(train_z) - train_y) * train_z)

    # 更新参数
    theta0 = tmp0
    theta1 = tmp1

    # 计算上一次的误差值
    current_error = E(train_z, train_y)  # 当前误差
    diff = error - current_error  # 差值
    error = current_error

    # 输出日志
    count += 1
    log = '第{}次：theta0 = {:.3f},theta1 = {:.3f},差值 = {:.4f}'
    print(log.format(count, theta0, theta1, diff))

x = np.linspace(-3, 3, 100)
# linspace(start, stop, num)
# start: 起始点，数组的第一个元素，包含在内。
# stop: 终止点，但不包含在数组中，即最后一个元素是 stop-1e-16（为了确保精确性，通常设置一个极小值）。
# num: 数组中的元素数量，包括起始和终止点，因此实际生成的数组长度为 num+1。

plt.plot(train_z, train_y, 'o')  # 展示训练数据
plt.plot(x, f(x), )  # 展示预测函数

print("______________")
# 验证
m1 = standardize(100)
n1 = f(standardize(100))
print(m1, n1)
m2 = standardize(200)
n2 = f(standardize(200))
print(m2, n2)
m3 = standardize(300)
n3 = f(standardize(300))
print(m3, n3)
plt.plot(m1, n1, "*", color="red")
plt.plot(m2, n2, "*", color="red")
plt.plot(m3, n3, "*", color="red")
plt.show()

    # 多项式回归

    '''
多项式回归
'''
import numpy as np
import matplotlib.pyplot as plt

# 读取文件
train = np.loadtxt('click.csv', delimiter=',', skiprows=1)
train_x = train[:, 0]  # 取出第一列
train_y = train[:, 1]  # 取出第二列
# plt.plot(train_x, train_y, "o")
# plt.show()

# 标准化(z-score规范化)
# 把训练数据变成平均值为0、方差为 1 的数据,加快参数收敛
mu = train_x.mean()  # mean() 给定数组的平均值
sigma = train_x.std()  # std() 标准差


# 标准化
def standardize(x):
    return (x - mu) / sigma


train_z = standardize(train_x)
# plt.plot(train_z, train_y, "o")
# plt.show()

# 参数初始化
theta = np.random.rand(3)  # 形状为3的一维数组 [x, y, z]


# 创建训练数据的矩阵
def to_matrix(x):
    return np.vstack([np.ones(x.shape[0]), x, x ** 2]).T  # 纵向合并，转置
# np.ones 创建全1数组
# shape[0]反映第0维度的大小，是矩阵的行数  1
# shape[1]反映第1维度的大小，是矩阵的列数


X = to_matrix(train_z)


# 预测函数
def f(x):
    return np.dot(x, theta)  # dot() 矩阵积


# 目标函数
def E(x, y):
    return 0.5 * np.sum(y - f(x)) ** 2


# 学习率10^-3
ETA = 1e-3

# 误差的差值
diff = 1

# 更新次数
count = 0

# 重复学习
error = E(X, train_y)  # 误差
while diff > 1e-2:
    # 更新参数
    theta = theta - ETA * np.dot(f(X) - train_y, X)

    # 计算上一次的误差值
    current_error = E(X, train_y)  # 当前误差
    diff = error - current_error  # 差值
    error = current_error

    # 输出日志
    count += 1
    log = '第 {} 次 : theta = {}, 差值 = {:.4f}'
    print(log.format(count, theta, diff))

x = np.linspace(-3, 3, 100)

plt.plot(train_z, train_y, 'o')  # 展示训练数据
plt.plot(x, f(to_matrix(x)), )  # 展示预测函数

print("______________")
# 验证
m1 = standardize(100)
k1 = to_matrix(np.array([m1]))
n1 = f(k1)
print(m1,n1)
m2 = standardize(200)
k2 = to_matrix(np.array([m2]))
n2 = f(k2)
print(m2,n2)
m3 = standardize(300)
k3 = to_matrix(np.array([m3]))
n3 = f(k3)
print(m3,n3)
plt.plot(m1, n1, "*", color="red")
plt.plot(m2, n2, "*", color="red")
plt.plot(m3, n3, "*", color="red")


plt.show()

    # 均方误差

    '''
均方误差
'''
import numpy as np
import matplotlib.pyplot as plt

# 读取文件
train = np.loadtxt('click.csv', delimiter=',', skiprows=1)
train_x = train[:, 0]  # 取出第一列
train_y = train[:, 1]  # 取出第二列

# 标准化(z-score规范化)
mu = train_x.mean()  # mean() 给定数组的平均值
sigma = train_x.std()  # std() 标准差


# 标准化
def standardize(x):
    return (x - mu) / sigma

train_z = standardize(train_x)

# 参数初始化
theta = np.random.rand(3)


# 创建训练数据的矩阵
def to_matrix(x):
    return np.vstack([np.ones(x.shape[0]), x, x ** 2]).T  # 纵向合并，转置

X = to_matrix(train_z)


# 预测函数
def f(x):
    return np.dot(x, theta)  # dot() 矩阵积

# 均方误差
def MSE(x,y):
    return (1/x.shape[0]) * np.sum((y - f(x))**2)

# 均方误差的历史记录
errors = []

# 学习率10^-3
ETA = 1e-3

# 误差的差值
diff = 1

# 更新次数
count = 0

# 重复学习
errors.append(MSE(X,train_y))
while diff > 1e-2:
    theta = theta - ETA * np.dot(f(X) - train_y, X)
    errors.append(MSE(X,train_y))
    diff = errors[-2]-errors[-1]

x = np.arange(len(errors))
# np.arange(start，stop，step)函数作用
# 1）一个参数时，参数值为终点，起点取默认值0，步长取默认值1。
# 2）两个参数时，第一个参数为起点，第二个参数为终点，步长取默认值1。
# 3）三个参数时，第一个参数为起点，第二个参数为终点，第三个参数为步长。其中步长支持小数
plt.plot(x, errors)
plt.show()

    # 随机梯度下降

    '''
随机梯度下降
'''

import numpy as np
import matplotlib.pyplot as plt

# 读取文件
train = np.loadtxt('click.csv', delimiter=',', skiprows=1)
train_x = train[:, 0]  # 取出第一列
train_y = train[:, 1]  # 取出第二列

# 标准化(z-score规范化)
mu = train_x.mean()  # mean() 给定数组的平均值
sigma = train_x.std()  # std() 标准差


# 标准化
def standardize(x):
    return (x - mu) / sigma


train_z = standardize(train_x)

# 参数初始化
theta = np.random.rand(3)


# 创建训练数据的矩阵
def to_matrix(x):
    return np.vstack([np.ones(x.shape[0]), x, x ** 2]).T  # 纵向合并，转置

X = to_matrix(train_z)
# print(X)  #  [[ 1.         -1.59139223  2.53252924].....]


# 预测函数
def f(x):
    return np.dot(x, theta)  # dot() 矩阵积


# 均方误差
def MSE(x, y):
    return (1 / x.shape[0]) * np.sum((y - f(x)) ** 2)


# 均方误差的历史记录
errors = []

# 学习率10^-3
ETA = 1e-3

# 误差的差值
diff = 1

# 更新次数
count = 0

# 重复学习
errors.append(MSE(X, train_y))
while diff > 1e-2:
    # 为了调整训练数据的顺序，准备随机的序列
    p = np.random.permutation(X.shape[0])
    # print(p)  # [3 5 1 11 15 4 9 16 14 13 8 0 2 19 7 12 10 6 18 17]  长度为20的随机序列
    # print(X.shape[0])  # 20,训练数据矩阵行数
    # 随机取出训练数据，使用随机梯度下降法更新参数
    for x, y in zip(X[p, :], train_y[p]):  # 使用zip()函数对随机选取的X和train_y进行迭代，每次迭代得到一对（x, y）
        theta = theta - ETA * (f(x) - y) * x
    # 计算与上一次的误差值
    errors.append(MSE(X, train_y))
    diff = errors[-2] - errors[-1]

    # 输出日志
    count += 1
    log = '第 {} 次 : theta = {}, 差值 = {:.4f}'
    print(log.format(count, theta, diff))

x = np.linspace(-3, 3, 100)
plt.plot(train_z,train_y,"o")
plt.plot(x,f(to_matrix(x)))
plt.show()
