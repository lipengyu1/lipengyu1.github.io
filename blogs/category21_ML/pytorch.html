    from torch.utils.data import Dataset
from PIL import Image
import os

class MyDataset(Dataset):

    def __init__(self,root_dir,label_dir):  # root_dir 数据集根目录 label_dir 标签目录
        self.root_dir = root_dir
        self.label_dir = label_dir
        self.path = os.path.join(self.root_dir,self.label_dir)  # 图片路径  将根目录和标签目录拼接成完整路径
        self.img_path = os.listdir(self.path)   # 使用 os.listdir 获取该路径下的所有图像文件名。

    def __getitem__(self, idx):  # idx 图片索引
        img_name = self.img_path[idx]  # 获取第 idx 个图片文件名
        img_item_path = os.path.join(self.root_dir,self.label_dir,img_name)  # 拼接图片完整路径
        img = Image.open(img_item_path)  # 打开图片
        label = self.label_dir  # 图片标签
        return img, label
    def __len__(self):
        return len(self.img_path)  # 返回数据集大小

root_dir = "dataset\\train"
ant_label_dir = "ants"
bees_label_dir = "bees"
ants_dataset = MyDataset(root_dir,ant_label_dir)
bees_dataset = MyDataset(root_dir,bees_label_dir)
train_dataset = ants_dataset + bees_dataset

    # 常见的transforms

    from PIL import Image
from torchvision import transforms
from torch.utils.tensorboard import SummaryWriter

writer =SummaryWriter("logs")
img_path = 'data/train/bees_image/17209602_fe5a5a746f.jpg'
img = Image.open(img_path)
print(img)
# <PIL.JpegImagePlugin.JpegImageFile image mode=RGB size=768x512 at 0x1DA43CD6D30>
#  size=768x512

# ToTensor的使用
# 用于将输入数据（例如图像、数组等）转换为张量的形式
trans_totensor = transforms.ToTensor()
img_tensor = trans_totensor(img)
writer.add_image("ToTensor", img_tensor)

# Normalize的使用
print(img_tensor[0][0][0])
# input[channel] = (input[channel] - mean[channel]) / std[channel]
trans_norm = transforms.Normalize([6, 3, 2], [9, 3, 5 ])
img_norm = trans_norm(img_tensor)
print(img_norm[0][0][0])
writer.add_image("Normalize", img_norm,2)

# Resize的使用
# 对输入图像进行大小调整
print(img.size)  # (500, 412)
trans_resize = transforms.Resize((512, 512))
# img PIL->resize->img_resize PIL
img_resize = trans_resize(img)
print(type(img_resize))
# <class 'PIL.Image.Image'>
print(img_resize)
# <PIL.Image.Image image mode=RGB size=512x512 at 0x1F88316C048>
# img_resize PIL->totensor->img_resize tensor
img_resize = trans_totensor(img_resize)
writer.add_image("Resize", img_resize,0)

# Compose -Resize -2 等比缩放
# 用于将多个图像预处理操作组合在一起
trans_resize2 = transforms.Resize(512)
# PIL-->PIL->tensor
trans_compose = transforms.Compose([trans_resize2, trans_totensor])
img_resize2 = trans_compose(img)
writer.add_image("Compose", img_resize2,1)

# RandomCrop的使用
# 用于从输入图像中随机裁剪出一块子图像
trans_random = transforms.RandomCrop(20)
trans_compose2 = transforms.Compose([trans_random, trans_totensor])
for i in range(10):
    img_crop = trans_compose2(img)
    writer.add_image("RandomCrop", img_crop, i)

writer.close()

    import torchvision
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

# 准备测试数据集
test_data = torchvision.datasets.CIFAR10(root='./dataset_cifar10', train=False,transform=torchvision.transforms.ToTensor())

test_loader = DataLoader(dataset=test_data, batch_size=64, shuffle=True, num_workers=0,drop_last=True)
# 参数说明：
# dataset：测试数据集
# batch_size：每批次的样本数
# shuffle：是否打乱数据集

# num_workers：多线程加载数据
# drop_last：是否丢弃最后一批不足batch_size的样本

# 测试数据集中第一张图片的大小和标签
img,target = test_data[0]
print(img.shape)  # torch.Size([3, 32, 32])
print(target)  # 3

wirter = SummaryWriter(log_dir='./logs_dataloader')
for epoch in range(2):
    step = 0
    for data in test_loader:
        imgs,targets = data

        # print(targets)  # tensor([9, 9, 5, 0]).....64张图片的标签
        wirter.add_images("Epoch:{}".format(epoch),imgs,step)
        step += 1

wirter.close()

    # 卷积

    import torch
import torch.nn.functional as F

# 定义输入张量和卷积核
# 5x5的输入张量
input = torch.tensor([[1, 2, 0, 3, 1],
                      [0, 1, 2, 3, 1],
                      [1, 2, 1, 0, 0],
                      [5, 2, 3, 1, 1],
                      [2, 1, 0, 1, 1]])   # torch.Size([5, 5])
# 3x3的卷积核
kernel = torch.tensor([[1, 2, 1],
                       [0, 1, 0],
                       [2, 1, 0]])  # torch.Size([3, 3])

# 将输入张量和卷积核调整为适合进行2D卷积的形状
# 即(batch_size, channels, height, width)
input = torch.reshape(input,(1,1,5,5))
# 1 表示 batch size（批量大小），这里只有一个样本。
# 1 表示 channels（通道数），这里只有一个通道。
# 5 表示 height（高度），即输入张量的高度为5。
# 5 表示 width（宽度），即输入张量的宽度为5。
kernel = torch.reshape(kernel,(1,1,3,3))

print(input.shape)  # torch.Size([1, 1, 5, 5])
print(kernel.shape)  # torch.Size([1, 1, 3, 3])

# 卷积 步长为1 输出结果是一个3x3的张量
# input：输入张量，形状为 (batch_size, in_channels, height, width)。
# weight：卷积核张量，形状为 (out_channels, in_channels, kernel_height, kernel_width)。
# bias：可选的偏置张量，形状为 (out_channels)。如果指定，将在卷积结果上加上偏置。
# stride：卷积步长，可以是一个整数或一个二元组 (stride_height, stride_width)。
# padding：输入张量的填充大小，可以是一个整数或一个二元组 (padding_height, padding_width)。
# dilation：卷积核元素之间的间距，可以是一个整数或一个二元组 (dilation_height, dilation_width)。
# groups：控制输入和输出之间的连接，in_channels 和 out_channels 必须能被 groups 整除。
output = F.conv2d(input,kernel,stride=1)
print(output)
# tensor([[[[10, 12, 12],
#           [18, 16, 16],
#           [13,  9,  3]]]])

# 步长为2  输出结果是一个2x2的张量
output2 = F.conv2d(input,kernel,stride=2)
print(output2)
# tensor([[[[10, 12],
#           [13,  3]]]])

# 步长为1，填充为1  输出结果是一个5x5的张量
output3 = F.conv2d(input,kernel,stride=1,padding=1)
print(output3)
# tensor([[[[ 1,  3,  4, 10,  8],
#           [ 5, 10, 12, 12,  6],
#           [ 7, 18, 16, 16,  8],
#           [11, 13,  9,  3,  4],
#           [14, 13,  9,  7,  4]]]])

    # 卷积层

    '''
卷积层
'''
import torch
import torchvision
from torch import nn
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

dataset = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=False, transform=torchvision.transforms.ToTensor(),download=True)

data_loader  = DataLoader(dataset, batch_size=64)

class TestNet(nn.Module):

    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(in_channels=3, out_channels=6, kernel_size=3,stride=1, padding=0)
    # in_channels (int): 输入图像的通道数。例如，对于彩色图像，这通常是 3（对应 RGB 通道）。
    # out_channels (int): 卷积层输出的通道数。这也是卷积层将学习的滤波器（核）的数量。
    # kernel_size (int 或 tuple): 卷积核的大小。如果提供一个整数，它将用于高度和宽度。如果提供一个元组，它将分别指定高度和宽度的核大小。
    # stride (int 或 tuple, 可选): 卷积的步幅。如果提供一个整数，它将用于高度和宽度。如果提供一个元组，它将分别指定高度和宽度的步幅。默认值是 1。
    # padding (int, tuple 或 str, 可选): 添加到输入的填充量。如果提供一个整数，它将用于高度和宽度。如果提供一个元组，它将分别指定高度和宽度的填充。如果提供一个字符串（'valid' 或 'same'），它将使用指定的填充策略。默认值是 0。
    # dilation (int 或 tuple, 可选): 核元素之间的间距。默认值是 1。
    # groups (int, 可选): 从输入通道到输出通道的阻塞连接数。默认值是 1。
    # bias (bool, 可选): 如果为 True，则向输出添加一个可学习的偏置。默认值是 True。
    def forward(self, x):
        x = self.conv1(x)
        return x

test_net = TestNet()
print(test_net)
# TestNet(
#   (conv1): Conv2d(3, 6, kernel_size=(3, 3), stride=(1, 1))
# )

step = 0
writer = SummaryWriter(log_dir='./nnlogs')
for data in data_loader:
    img, label = data
    output = test_net(img)
    print(img.shape)
    print(output.shape)
    writer.add_images("imput", img, step)
    output = torch.reshape(output,(-1,3,30,30))
    # -1表示这个维度的大小将由PyTorch自动计算，以确保总元素数量不变。
    # 3表示新的张量在这个维度上有3个通道。
    # 30表示新的张量在空间维度上的高度和宽度都是30。
    writer.add_images("output", output, step)
    step += 1
writer.close()

    # 池化层——最大池化

    '''
最大池化层
最大池化层（Max Pooling Layer）是一种池化层
它会在输入张量的每个窗口（kernel_size大小的窗口）中找到最大值，并输出该窗口中的最大值。
'''
import torch
from torch import nn
from torch.nn import MaxPool2d

input = torch.tensor([[1, 2, 0, 3, 1],
                      [0, 1, 2, 3, 1],
                      [1, 2, 1, 0, 0],
                      [5, 2, 3, 1, 1],
                      [2, 1, 0, 1, 1]],dtype=torch.float32)   # torch.Size([5, 5])

input = torch.reshape(input, (-1, 1, 5, 5))
# -1表示第一维的大小将根据张量的总元素数量和其他维度的大小自动计算
print(input.shape)
#

class TestNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.maxpool1 = MaxPool2d(kernel_size = 3,ceil_mode = True)
        # kernel_size：池化窗口的大小。可以是一个整数（表示正方形窗口的边长）或一个元组（表示矩形窗口的高度和宽度）。
        # stride：池化窗口的步幅。默认值为 kernel_size，即池化窗口的大小。
        # padding：在输入的每一边添加的零填充的数量。默认值为 0。
        # dilation：控制池化窗口中元素之间的间隔。默认值为 1。
        # return_indices：如果为 True，则返回最大值的索引。这在反池化操作中很有用。默认值为 False。
        # ceil_mode：如果为 True，则使用向上取整而不是向下取整来计算输出形状。默认值为 False。

    def forward(self, x):
        output = self.maxpool1(x)
        return output

testnet = TestNet()
output =  testnet(input)
print(output)
# tensor([[[[2., 3.],
#           [5., 1.]]]])

import torch
import torchvision
from torch import nn
from torch.nn import Conv2d, MaxPool2d, Linear, Sequential
from torch.nn.modules.flatten import Flatten
from torch.utils.data import DataLoader

dataset = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=False, transform=torchvision.transforms.ToTensor(),download=True)
dataloader = DataLoader(dataset, batch_size=1)
class Net(nn.Module):
    def __init__(self):
        super().__init__()
        self.model1 = Sequential(
            Conv2d(3, 32, 5, padding=2),
            MaxPool2d(2),
            Conv2d(32, 32, 5, padding=2),
            MaxPool2d(2),
            Conv2d(32, 64, 5, padding=2),
            MaxPool2d(2),
            Flatten(),
            Linear(1024, 64),
            Linear(64, 10)
        )

    def forward(self, x):

        x = self.model1(x)
        return x

# 定义损失函数和网络实例
loss = nn.CrossEntropyLoss()
net = Net()
# 优化器
optim = torch.optim.SGD(net.parameters(), lr=0.01)
for epoch in range(20): # 训练20轮
    running_loss = 0.0
    for data in dataloader:
        imgs,target = data
        output = net(imgs)
        result_loss = loss(output, target)
        optim.zero_grad()  # 梯度清零
        result_loss.backward()  # 反向传播，计算梯度
        optim.step()  # 更新参数
        # 累加每个batch的损失，用于记录整个epoch的总损失
        runnning_loss = result_loss+running_loss
    print(runnning_loss)

    import torch
import torchvision
from torch import nn
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

from model import Net

# 准备数据集
# 训练集
train_data = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=True,transform=torchvision.transforms.ToTensor(), download=True)
# 测试集
test_data = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=False,transform=torchvision.transforms.ToTensor(), download=True)
# 训练集长度
train_data_size = len(train_data)
print("训练集长度为：{}".format(train_data_size))
# 测试集长度
test_data_size = len(test_data)
print("测试集长度为：{}".format(test_data_size))

# 准备数据加载器
train_dataloader = DataLoader(train_data, batch_size=64)
test_dataloader = DataLoader(test_data, batch_size=64)

# 创建网络模型
net = Net()

# 定义损失函数
loss_fn = nn.CrossEntropyLoss()

# 定义优化器
learning_rate = 1e-2  # 学习速率
optimizer = torch.optim.SGD(net.parameters(), lr=learning_rate)

# 设置训练网络参数
# 记录训练次数
total_train_step = 0
# 记录测试次数
total_test_step = 0
# 训练轮数
epoch = 10
# 添加tensorboard
writer = SummaryWriter('log_train')

for i in range(epoch):
    print("第{}轮训练开始...".format(i+1))
    # 训练步骤开始
    net.train()
    for data in train_dataloader:
        imgs, targets = data
        outputs = net(imgs)
        # 计算损失
        loss = loss_fn(outputs, targets)
        # 优化器优化模型
        optimizer.zero_grad()
        # 反向传播
        loss.backward()
        optimizer.step()

        total_train_step += 1
        if total_train_step % 100 == 0:
            print("训练次数：{}，损失：{:.4f}".format(total_train_step, loss.item()))
            writer.add_scalar('train_loss', loss.item(), total_train_step)

    # 测试步骤开始
    net.eval()
    total_test_loss = 0
    # 正确个数
    total_accuracy = 0
    with torch.no_grad():
        for data in test_dataloader:
            imgs, targets = data
            outputs = net(imgs)
            loss = loss_fn(outputs, targets)
            total_test_loss +=loss.item() # 测试集损失
            accuracy = (outputs.argmax(1) == targets).sum().item()
            total_accuracy += accuracy  # 测试集正确个数
    print("整体测试集损失：{:.4f}".format(total_test_loss))
    print("整体测试集上的正确率：{:.4f}".format(total_accuracy / test_data_size))
    writer.add_scalar('test_loss', total_test_loss,total_test_step)
    writer.add_scalar('accuracy', total_accuracy / test_data_size, total_test_step)
    total_test_step = total_test_step + 1

    # 保存模型
    torch.save(net,"net_{}.pth".format(i))
    # torch.save(net.state_dict(),"net_{}.pth".format(i))
    print("模型已保存")
writer.close()

    使用if torch.cuda.is_available(): XXX = XXX.cuda()

    import torch
import torchvision
from torch import nn
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

from model import Net

# 准备数据集
# 训练集
train_data = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=True,
                                          transform=torchvision.transforms.ToTensor(), download=True)
# 测试集
test_data = torchvision.datasets.CIFAR10(root='dataset_cifar10', train=False,
                                         transform=torchvision.transforms.ToTensor(), download=True)
# 训练集长度
train_data_size = len(train_data)
print("训练集长度为：{}".format(train_data_size))
# 测试集长度
test_data_size = len(test_data)
print("测试集长度为：{}".format(test_data_size))

# 准备数据加载器
train_dataloader = DataLoader(train_data, batch_size=64)
test_dataloader = DataLoader(test_data, batch_size=64)


# 定义网络结构
class Net(nn.Module):

    def __init__(self):
        super(Net, self).__init__()
        self.model = nn.Sequential(
            nn.Conv2d(3, 32, 5, 1, padding=2),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 32, 5, 1, 2),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 64, 5, 1, 2),
            nn.MaxPool2d(2),
            nn.Flatten(),
            nn.Linear(64 * 4 * 4, 64),
            nn.Linear(64, 10)
        )

    def forward(self, x):
        x = self.model(x)
        return x

# 网络模型，数据，损失函数-->cuda()
# 创建网络模型
net = Net()
if torch.cuda.is_available():
    net = net.cuda()

# 定义损失函数
loss_fn = nn.CrossEntropyLoss()
if torch.cuda.is_available():
    loss_fn = loss_fn.cuda()
# 定义优化器
learning_rate = 1e-2  # 学习速率
optimizer = torch.optim.SGD(net.parameters(), lr=learning_rate)

# 设置训练网络参数
# 记录训练次数
total_train_step = 0
# 记录测试次数
total_test_step = 0
# 训练轮数
epoch = 10
# 添加tensorboard
writer = SummaryWriter('log_train')

for i in range(epoch):
    print("第{}轮训练开始...".format(i + 1))
    # 训练步骤开始
    net.train()
    for data in train_dataloader:
        imgs, targets = data
        if torch.cuda.is_available():
            imgs = imgs.cuda()
            targets = targets.cuda()
        outputs = net(imgs)
        # 计算损失
        loss = loss_fn(outputs, targets)
        # 优化器优化模型
        optimizer.zero_grad()
        # 反向传播
        loss.backward()
        optimizer.step()

        total_train_step += 1
        if total_train_step % 100 == 0:
            print("训练次数：{}，损失：{:.4f}".format(total_train_step, loss.item()))
            writer.add_scalar('train_loss', loss.item(), total_train_step)

    # 测试步骤开始
    net.eval()
    total_test_loss = 0
    # 正确个数
    total_accuracy = 0
    with torch.no_grad():
        for data in test_dataloader:
            imgs, targets = data
            if torch.cuda.is_available():
                imgs = imgs.cuda()
                targets = targets.cuda()
            outputs = net(imgs)
            loss = loss_fn(outputs, targets)
            total_test_loss += loss.item()  # 测试集损失
            accuracy = (outputs.argmax(1) == targets).sum().item()
            total_accuracy += accuracy  # 测试集正确个数
    print("整体测试集损失：{:.4f}".format(total_test_loss))
    print("整体测试集上的正确率：{:.4f}".format(total_accuracy / test_data_size))
    writer.add_scalar('test_loss', total_test_loss, total_test_step)
    writer.add_scalar('accuracy', total_accuracy / test_data_size, total_test_step)
    total_test_step = total_test_step + 1

    # 保存模型
    torch.save(net, "net_{}.pth".format(i))
    # torch.save(net.state_dict(),"net_{}.pth".format(i))
    print("模型已保存")
writer.close()
