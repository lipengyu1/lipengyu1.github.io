    # SQL语言

    # 单表查询

    -- 查询表中所有记录
SELECT name,age ,score,addtime FROM stu_info;

-- 对结果中的列进行重命名
SELECT name,age,score,addtime FROM stu_info;

-- 查询姓名、年龄、对结果集进行去重
SELECT DISTINCT	name ,age FROM stu_info;

-- selsect后面可以跟表达式
SELECT name ,age + 20,score ,score -20 FROM stu_info;

-- 查询年龄大于18的学生
SELECT * FROM stu_info WHERE age > 18;
SELECT * FROM stu_info WHERE age < 18;<b
SELECT * FROM stu_info WHERE age = 18;<br>

-- 查询年龄 = 18，性别为男
SELECT * FROM stu_info WHERE age = 18 AND sex = '男';

-- 查询女生成绩>95
SELECT name,age,sex,score FROM stu_info WHERE score > 95 AND sex = '女';

-- or
SELECT * FROM stu_info WHERE age = 18 OR sex = '男';

-- BETWEEN AND
SELECT name,age,sex,score,addtime FROM stu_info WHERE addtime BETWEEN '2017-09-07 00:00:00' AND '2017-09-08 23:50:59'
SELECT * FROM stu_info WHERE addtime >= '2017-09-07 00:00:00' AND addtime <= '2017-09-08 <br>23:50:59'

-- 查询年龄在18-20
SELECT * FROM stu_info WHERE age BETWEEN 18 AND 20;
SELECT * FROM stu_info WHERE age >= 18 AND age <=20;

-- 查询1,2,3学生
SELECT * FROM stu_info WHERE gid = 1 OR gid = 2 OR gid = 3;

-- in / not in 表示某个字段值在某个集合中
SELECT * FROM stu_info WHERE gid NOT IN (1,2,3); 

-- 模糊查询
-- 查询名字以李开头 
-- 模糊查询 like 后加通配符 %代表n个字符 
SELECT * FROM stu_info WHERE name LIKE '李%'; 
-- 名字为李XX 
SELECT * FROM stu_info WHERE name LIKE '李'
-- 名字中有文
SELECT * FROM stu_info WHERE name LIKE '%文%';
-- 名字有两个字
SELECT * FROM stu_info WHERE name LIKE '__' 
-- 名字至少有两个字
SELECT * FROM stu_info WHERE name LIKE '__%'；

-- 对班上男生成绩进行排名,ORDER BY 默认进行升序排列，降序加 DESC
-- order by可以加多个条件
SELECT * FROM stu_info WHERE sex = '男'
ORDER BY score DESC,age DESC;

-- 聚合函数 将一系列数据进行聚合求值
SELECT SUM( score ) FROM stu_info;
SELECT AVG( score ) FROM stu_info;
SELECT MAX( score ) FROM stu_info;
SELECT MIN( score ) FROM stu_info;
SELECT MAX( age ) FROM stu_info;
SELECT AVG( age ) FROM stu_info;

-- 求一组学生成绩的平均值、最大值
SELECT AVG(score) 平均值, MAX(score) 最大值 FROM stu_info WHERE gid = 1;

-- 统计一组里的学生人数
SELECT COUNT(id) FROM stu_info WHERE gid = 1;
-- 统计女生总人数、年龄最大值、平均成绩(round ,近似值)
SELECT count(*) as 总人数,ROUND(AVG(score),2) as 平均成绩,MAX(age) 最大年龄
FROM stu_info WHERE sex = '女';

-- 按组号分组
SELECT gid,COUNT(*) 组人数,MAX(score) 成绩最大值,MIN(score) 成绩最小值,AVG(age) 年龄平均值
FROM stu_info GROUP BY gid;

-- 统计男生女生的各自人数和平均成绩,成绩最大值
SELECT sex,COUNT(*) 人数,MAX(score) 成绩最大值,MIN(score) 成绩最小值,AVG(score) 成绩平均值
FROM stu_info GROUP BY sex;

-- 统计每组中男生、女生人数
SELECT gid 组号,sex 性别,COUNT(*)
FROM stu_info
GROUP BY gid,sex;

-- HAVING ：对汇总查询后的结果在此进行过滤HAVING 只能跟随GROUP BY 使用

-- 统计男生、女生各自的人数，并将人数>5的性别详情打印
SELECT COUNT(*) ,sex FROM stu_info
GROUP BY sex
HAVING COUNT(*) > 5;

-- 更新表中的记录
UPDATE stu_info SET age = 20
WHERE id = 10015;

UPDATE  stu_info SET sex='女'
WHERE id = 10015;

UPDATE  stu_info SET age = age+2
WHERE id > 10015;

-- 删除表中的记录
DELETE FROM stu_info WHERE id = 10014;

-- 分页查询
SELECT * FROM stu_info LIMIT 10;
SELECT * FROM stu_info LIMIT 1,5;

    # 多表查询

    -- 查询出每个部门的员工数、平均年龄
SELECT dname,COUNT(*),ROUND(AVG(age)) FROM dep,employee
WHERE employee.depid=dep.id
GROUP BY dname;

-- 查询部门中人数大于1的部门
SELECT dname,COUNT(*)FROM dep,employee
WHERE employee.depid=dep.id
GROUP BY dname
HAVING COUNT(*)>1;


-- 查询每个部门中不同员工的人数
SELECT dname,age,count(*) FROM dep,employee
WHERE employee.depid=dep.id
GROUP BY dname,age;

-- 查询员工、部门详细信息
SELECT * FROM employee,dep
WHERE employee.depid=dep.id;

-- 内连接
SELECT * FROM employee INNER JOIN dep
ON employee.depid=dep.id;

-- 左连接,以左表为主
SELECT * FROM employee LEFT JOIN dep
ON employee.depid=dep.id;

-- 右连接，以右表为主
SELECT * FROM employee RIGHT JOIN dep
ON employee.depid=dep.id;

    # 作业

    -- 选择部门30中所有的员工
SELECT * FROM emp WHERE deptno = 30;
-- 列出所有业务人员（CLERK）的姓名，编号，和部门编号
SELECT	ename,empno,deptno  FROM emp WHERE job = 'CLERK';
-- 找出佣金高于薪水的员工
SELECT * FROM emp WHERE comm > sal;
-- 找出佣金高于薪水60%的员工
SELECT * FROM emp WHERE comm > sal*0.6;
-- 找出部门10中所有的经理和部门20中所有业务员的详细资料
SELECT * FROM emp WHERE (deptno = 10 AND job = 'MANAGER') OR (deptno = 20 AND job = 'CLERK');
-- 既不是经理又不是业务员但薪水大于等于2000的所有的员工信息
SELECT * FROM emp	WHERE sal > 2000 AND job NOT IN ('MANAGER' ,'CLERK');
-- 找出收取佣金的员工的不同工作
SELECT DISTINCT job FROM emp WHERE comm != 'NULL' ;
-- 显示正好为5个字符的员工的姓名
SELECT ename FROM emp WHERE ename LIKE '_____';
-- 显示不带有"R"的员工的姓名
SELECT ename FROM emp WHERE ename NOT LIKE '%R%';
-- 显示所有员工姓名的前三个字符
SELECT SUBSTR(ename,1,3) FROM emp;
-- 显示所有员工的姓名,用a替换所有A
SELECT ename ,REPLACE(ename,'A','a') FROM emp;
-- 显示员工的详细资料,按姓名排序
SELECT * FROM emp 
ORDER BY ename;
-- 找出各月倒数第3天受雇的所有员工,LAST_DAY() 获取当前日期最后一天
SELECT ename,hiredate,LAST_DAY(hiredate)-2 FROM emp WHERE hiredate= LAST_DAY(hiredate)-2;
-- 找出早于35年前受雇的员工
SELECT hiredate,ename FROM emp WHERE YEAR(hiredate) < YEAR( CURRENT_DATE)-35 ;
-- 以字母小写的方式显示所有员工的姓名
SELECT ename ,LOWER(ename) FROM emp;
-- 显示员工的姓名和受雇日期,根据其服务年限,将最老的员工排在最前面
SELECT ename,hiredate
FROM emp 
order by hiredate;
-- 显示所有员工的姓名、工作和薪金,按工作的降序排,若工作相同则按薪金排序
SELECT ename,job,sal FROM emp 
ORDER BY job DESC,sal DESC;
-- 显示所有员工的姓名、加入公司的年份和月份,按受雇日期所在月排序,若月份相同则将最早年份的员工排在最前面
SELECT ename,hiredate,YEAR(hiredate)FROM emp ORDER BY MONTH(hiredate),YEAR(hiredate);
-- 找出在(任何年份的)2月受聘的所有员工
SELECT ename ,empno ,hiredate FROM emp WHERE MONTH(hiredate)=2;
-- 对于每个员工,显示其加入公司的天数  DATEDIFF计算两个日期的相差天数
SELECT ename,hiredate,DATEDIFF(CURRENT_DATE,hiredate) FROM emp;
-- 显示姓名字段的任何位置包含"A"的所有员工的姓名
SELECT ename FROM emp WHERE ename LIKE '%A%';
-- 列出至少有2个员工的部门名称
SELECT * FROM emp,dept WHERE emp.deptno = dept.deptno
GROUP BY dname 
HAVING COUNT(*)>2;

    # 子查询

    -- 查询所有电子产品的平均价格，保留两位小数
SELECT ROUND(AVG(goods_price),2) FROM tdb_goods;
-- 查询出所有价格大于平均价格的商品，并降序排列
SELECT * FROM tdb_goods
WHERE goods_price>
		(SELECT ROUND(AVG(goods_price),2)
		FROM tdb_goods)
		ORDER BY  DESC;
		
-- 查询超极本价格
SELECT goods_price FROM tdb_goods
WHERE goods_cate = '超级本'
-- 查询价格大于超级本价格的电子商品(大于最小值)
SELECT * FROM tdb_goods
WHERE goods_price> ANY
		(SELECT goods_price FROM tdb_goods
		WHERE goods_cate = '超级本')
-- 或
SELECT * FROM tdb_goods
WHERE goods_price> SOME
		(SELECT goods_price FROM tdb_goods
		WHERE goods_cate = '超级本')
-- 大于最大值
SELECT * FROM tdb_goods
WHERE goods_price> ALL
		(SELECT goods_price FROM tdb_goods
		WHERE goods_cate = '超级本')

    cd /var/lib/mysql
rm -rf auto.cnf
systemctl restart  mysqld
